// BallMovement
using gwyfhelper;
public void PreUpdate()
{
  GoFast.Update(this.hole, this.hitForce, this.preHitLocation, base.transform, this.rb, this.playerCamPivot, this.currentVelocity, this.minVelToHit, this.outOfBounds, this.onTrolley, this.initialDrag, this.sandDragToApply, this.waterDragToApply);
  this.hole = GoFast.GetNewHole();
  this.playerCustomProps["holeNumber"] = this.hole;
  this.currentHoleNumber = this.hole;
  this.hitForce = GoFast.GetNewHitForce();
  this.preHitLocation = GoFast.GetNewPreHitLocation();
  if (GoFast.ShouldResetToLastShot())
  {
    this.ResetToLastShot();
  }
}
